AUTH
Routes:
1. /auth/signup              (POST)
2. /auth/login               (POST)
3. /auth/forgot-password     (POST)
4. /auth/otp                 (POST)

POSTS
Route:
1. /post                   -- fetch all posts   (GET)
2. /post                   -- create post       (POST)
1. /post/:id               -- fetch particular post (GET)
2. /post/delete/:id        -- delete post   (DELETE)
3. /post/edit/:id          -- edit post     (PUT)
4. /post/like/:id          -- like post     (PUT)

USER
1. /users                   -- fetch all users  (GET)
2. /user                    -- fetch current user info  (GET)
3. /user                    -- update current user info  (PATCH)
4. /user                    -- delete current user  (DELETE)
5. /user/:id                -- fetch particular user info   (GET)
6. /user/:id                -- update particular user info  (PATCH)
7. /user/:id                -- delete particular user  (DELETE)


USER OBJECT: {
    _id: String;         (use ObjectId)
    first_name: String;
    last_name: String;
    email: String;
    username: String;  
	created_at: Date;
}

POST OBJECT: {
    _id: String;     (use ObjectId)
    title: String;
    body: String;
    image?: String; (url)
    video?: String;
    createdAt: Date;
    modifiedAt: Date;
    creator: User;
}
